CREATE TABLE IF NOT EXISTS todos (
    id SERIAL PRIMARY KEY,
    title VARCHAR(255) NOT NULL,
    description TEXT,
    completed BOOLEAN DEFAULT FALSE
);


# In terminal 1: Produce messages:
sudo docker exec -it kafka-1 bash -c "while true; do echo \$(date) | kafka-console-producer.sh --broker-list kafka-1:19092 --topic todo_created; sleep 1; done"

# In terminal 2: Start consumer:
sudo docker exec -it kafka-1 kafka-console-consumer.sh \
  --bootstrap-server kafka-1:19092 \
  --topic todo_created \
  --from-beginning
